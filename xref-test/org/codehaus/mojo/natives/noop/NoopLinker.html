1   package org.codehaus.mojo.natives.noop;

3   import java.io.File;
4   import java.util.List;
5   import org.codehaus.mojo.natives.NativeBuildException;
6   import org.codehaus.mojo.natives.linker.Linker;
7   import org.codehaus.mojo.natives.linker.LinkerConfiguration;
8   import org.codehaus.plexus.component.annotations.Component;
9   
10  @Component(role = Linker.class, hint = "noop")
11  public class NoopLinker
12      implements Linker
13  {
14  
15      @Override
16      public File link( LinkerConfiguration config, List<File> compilerOutputFiles )
17          throws NativeBuildException
18      {
19          String fileName = config.getOutputFileName();
20  
21          if ( config.getOutputFileExtension() != null )
22          {
23              fileName += "." + config.getOutputFileExtension();
24          }
25          return new File( config.getOutputDirectory(), fileName );
26      }
27  
28  }
