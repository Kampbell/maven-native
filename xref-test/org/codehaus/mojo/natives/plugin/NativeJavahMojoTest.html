19      public void testClassPathBeginsWithProjectOutputDirectory()
20          throws Exception
21      {
22          File pluginXml = new File( getBasedir(), "src/test/resources/javah/plugin-config.xml" );
23          NativeJavahMojo mojo = (NativeJavahMojo) this.lookupMojo( "javah", pluginXml );
24  
25          mojo.execute();
26  
27          JavahConfiguration config = mojo.getJavahConfiguration();
28  
29          // only found ${project.build.outputDirectory}
30          assertEquals( 1, config.getClassPaths().length );
31  
32          String classesDir = config.getClassPaths()[0];
33  
34          assertEquals( mojo.getProject().getBuild().getOutputDirectory(), classesDir );
35  
36          assertEquals( new File( "target/javah" ), config.getOutputDirectory() );
37  
38          assertEquals( new File( getBasedir() ), config.getWorkingDirectory() );
39  
40          // in test mode, the project starts out of 0 source root instead of 1
41          assertEquals( 1, mojo.getProject().getCompileSourceRoots().size() );
42  
43      }
