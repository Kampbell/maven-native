199     @Override
200     public void execute()
201         throws MojoExecutionException
202     {
203 
204         if ( StringUtils.isEmpty( this.classifier ) )
205         {
206             this.classifier = null;
207         }
208 
209         Linker linker = this.getLinker();
210 
211         this.config = this.createLinkerConfiguration();
212 
213         try
214         {
215             List<File> allCompilerOuputFiles = this.getAllCompilersOutputFileList();
216 
217             File outputFile = linker.link( config, allCompilerOuputFiles );
218             allCompilerOuputFiles.clear();
219 
220             // to be used by post linker mojo like native:manifest
221             @SuppressWarnings({ "unused", "unchecked" })
222             Object unchecked = this.getPluginContext().put( AbstractNativeMojo.LINKER_OUTPUT_PATH, outputFile );
223 
224         }
225         catch ( IOException ioe )
226         {
227             throw new MojoExecutionException( ioe.getMessage(), ioe );
228         }
229         catch ( NativeBuildException nbe )
230         {
231             throw new MojoExecutionException( nbe.getMessage(), nbe );
232         }
233 
234         if ( this.attach )
235         {
236             this.attachPrimaryArtifact();
237 
238             this.attachSecondaryArtifacts();
239         }
240     }

242     private LinkerConfiguration createLinkerConfiguration()
243         throws MojoExecutionException
244     {
245         LinkerConfiguration config = new LinkerConfiguration();
246         config.setWorkingDirectory( this.workingDirectory );
247         config.setExecutable( this.linkerExecutable );
248         config.setStartOptions( removeEmptyOptions( this.linkerStartOptions ) );
249         config.setMiddleOptions( removeEmptyOptions( this.linkerMiddleOptions ) );
250         config.setEndOptions( removeEmptyOptions( this.linkerEndOptions ) );
251         config.setOutputDirectory( this.linkerOutputDirectory );
252         config.setOutputFileName( this.linkerFinalName );
253         config.setOutputFileExtension( this.linkerFinalNameExt );
254         config.setExternalLibDirectory( this.externalLibDirectory );
255         config.setExternalLibFileNames( this.getLibFileNames() );
256         config.setEnvFactory( this.getEnvFactory() );
257         config.setUsingLinkerResponseFile( usingLinkerResponseFile );
258         config.setCheckStaleLinkage( this.checkStaleLinkage );
259 
260         return config;
261     }

263     private Linker getLinker()
264         throws MojoExecutionException
265     {
266         Linker linker;
267 
268         try
269         {
270             if ( this.linkerProvider == null )
271             {
272                 this.linkerProvider = this.compilerProvider;
273             }
274 
275             linker = this.manager.getLinker( this.linkerProvider );
276         }
277         catch ( NoSuchNativeProviderException pe )
278         {
279             throw new MojoExecutionException( pe.getMessage() );
280         }
281 
282         return linker;
283     }

285     /**
286      *
287      */
288     private void attachPrimaryArtifact()
289     {
290         Artifact artifact = this.project.getArtifact();
291 
292         if ( null == this.classifier )
293         {
294             artifact.setFile( new File( this.linkerOutputDirectory + "/" + this.project.getBuild().getFinalName() + "."
295                     + this.project.getArtifact().getArtifactHandler().getExtension() ) );
296         }
297         else
298         {
299             // install primary artifact as a classifier
300 
301             DefaultArtifact clone = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
302                     artifact.getVersionRange().cloneOf(), artifact.getScope(), artifact.getType(), classifier,
303                     artifact.getArtifactHandler(), artifact.isOptional() );
304 
305             clone.setRelease( artifact.isRelease() );
306             clone.setResolvedVersion( artifact.getVersion() );
307             clone.setResolved( artifact.isResolved() );
308             clone.setFile( artifact.getFile() );
309 
310             if ( artifact.getAvailableVersions() != null )
311             {
312                 clone.setAvailableVersions( new ArrayList<>( artifact.getAvailableVersions() ) );
313             }
314 
315             clone.setBaseVersion( artifact.getBaseVersion() );
316             clone.setDependencyFilter( artifact.getDependencyFilter() );
317 
318             if ( artifact.getDependencyTrail() != null )
319             {
320                 clone.setDependencyTrail( new ArrayList<>( artifact.getDependencyTrail() ) );
321             }
322 
323             clone.setDownloadUrl( artifact.getDownloadUrl() );
324             clone.setRepository( artifact.getRepository() );
325 
326             clone.setFile( new File( this.linkerOutputDirectory + "/" + this.project.getBuild().getFinalName() + "."
327                     + this.project.getArtifact().getArtifactHandler().getExtension() ) );
328 
329             project.setArtifact( clone );
330         }
331     }

333     private void attachSecondaryArtifacts()
334     {
335         final String[] tokens;
336         if ( this.linkerSecondaryOutputExtensions != null )
337         {
338             tokens = StringUtils.split( this.linkerSecondaryOutputExtensions, "," );
339         }
340         else
341         {
342             tokens = new String[0];
343         }
344 
345         for ( int i = 0; i < tokens.length; ++i )
346         {
347             // TODO: shouldn't need classifier
348             Artifact artifact = artifactFactory.createArtifact( project.getGroupId(), project.getArtifactId(),
349                     project.getVersion(), this.classifier, tokens[i].trim() );
350             artifact.setFile( new File( this.linkerOutputDirectory + "/" + this.project.getBuild().getFinalName() + "."
351                     + tokens[i].trim() ) );
352 
353             project.addAttachedArtifact( artifact );
354         }
355

358     private List<String> getLibFileNames()
359         throws MojoExecutionException
360     {
361         List<String> libList = new ArrayList<>();
362 
363         Set<Artifact> artifacts = this.project.getArtifacts();
364 
365         for ( Iterator<Artifact> iter = artifacts.iterator(); iter.hasNext(); )
366         {
367             Artifact artifact = iter.next();
368 
369             if ( INCZIP_TYPE.equals( artifact.getType() ) )
370             {
371                 continue;
372             }
373 
374             String libFileName = FileUtils.filename( this.getDependencyFile( artifact, true ).getPath() );
375 
376             libList.add( libFileName );
377         }
378 
379         libList = this.reorderLibDependencies( libList );
380 
381         return libList;
382     }

384     /**
385      * convert dependencyLinkingOrders to a file list
386      *
387      * @return
388      */
389     private List<String> getDependenciesFileOrderList()
390         throws MojoExecutionException
391     {
392         List<String> list = new ArrayList<>();
393 
394         if ( this.linkingOrderLibs != null )
395         {
396             for ( Iterator<String> i = linkingOrderLibs.iterator(); i.hasNext(); )
397             {
398                 String element = i.next();
399 
400                 Artifact artifact = lookupDependencyUsingGroupArtifactIdPair( element );
401 
402                 if ( artifact != null )
403                 {
404                     String libFileName = FileUtils.filename( this.getDependencyFile( artifact, false ).getPath() );
405 
406                     list.add( libFileName );
407                 }
408                 else
409                 {
410                     throw new MojoExecutionException( element + " not found on project dependencies." );
411                 }
412             }
413         }
414 
415         return list;
416     }

418     /**
419      * Look up library in dependency list using groupId:artifactId key Note: we can not use project.artifactMap due the
420      * introduction of inczip dependency where 2 dependency with the same artifactId and groupId, but differs by
421      * extension type make the map not suitable for lookup
422      *
423      * @param groupArtifactIdPair
424      * @return
425      * @throws MojoExecutionException
426      */
427     private Artifact lookupDependencyUsingGroupArtifactIdPair( String groupArtifactIdPair )
428         throws MojoExecutionException
429     {
430         String[] tokens = StringUtils.split( groupArtifactIdPair, ":" );
431 
432         if ( tokens.length != 2 )
433         {
434             throw new MojoExecutionException( "Invalid groupId and artifactId pair: " + groupArtifactIdPair );
435         }
436 
437         Set<Artifact> allDependencyArtifacts = project.getDependencyArtifacts();
438 
439         for ( Iterator<Artifact> iter = allDependencyArtifacts.iterator(); iter.hasNext(); )
440         {
441             Artifact artifact = iter.next();
442             if ( INCZIP_TYPE.equals( artifact.getType() ) )
443             {
444                 continue;
445             }
446 
447             if ( tokens[0].equals( artifact.getGroupId() ) && tokens[1].equals( artifact.getArtifactId() ) )
448             {
449                 return artifact;
450             }
451         }
452 
453         return null;
454 
455     }

457     private List<String> reorderLibDependencies( List<String> libs )
458         throws MojoExecutionException
459     {
460         List<String> requestedOrderList = getDependenciesFileOrderList();
461 
462         if ( requestedOrderList.size() != 0 )
463         {
464             // remove from original list first
465             for ( Iterator<String> i = requestedOrderList.iterator(); i.hasNext(); )
466             {
467                 libs.remove( i.next() );
468             }
469 
470             for ( Iterator<String> i = libs.iterator(); i.hasNext(); )
471             {
472                 requestedOrderList.add( i.next() );
473             }
474 
475             return requestedOrderList;
476         }
477         else
478         {
479             return libs;
480         }
481     }

483     private File getDependencyFile( Artifact artifact, boolean doCopy )
484         throws MojoExecutionException
485     {
486 
487         File newLocation = new File( this.externalLibDirectory,
488                 artifact.getArtifactId() + "." + artifact.getArtifactHandler().getExtension() );
489 
490         try
491         {
492             if ( doCopy && !artifact.getFile().isDirectory()
493                     && ( !newLocation.exists() || newLocation.lastModified() <= artifact.getFile().lastModified() ) )
494             {
495                 FileUtils.copyFile( artifact.getFile(), newLocation );
496             }
497         }
498         catch ( IOException ioe )
499         {
500             throw new MojoExecutionException( "Unable to copy dependency to staging area.  Could not copy "
501                     + artifact.getFile() + " to " + newLocation, ioe );
502         }
503 
504         return newLocation;
505     }
